async def scrape_opportunities():
    """Scrape grant/loan/crowdfunding opportunities using Zyte."""

                opportunities.append({
                    'name': opp['name'],
                    'url': opp['url'],
                    'type': opp['type'],
                    'details': title
                })

            except Exception as e:
                logger.error(f"Error scraping {opp['url']}: {str(e)}")

        await browser.close()
